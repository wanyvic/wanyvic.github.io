match 匹配

let some_number = Some(5);
let some_string = Some("string");
let undefined_number: Option<i32> = None;
// 必须穷举所有可能
match some_number {
    Some(s) => {
        println!("{}", s + 1);
        Some(s + 1)
    }
    None => None,
};
// - => () 通配符
match some_number {
    Some(3) | Some(5) => println!("3 or 5"),
    Some(x) => println!("{}", x),
    _ => println!("None"),
};

if let 匹配

if let [类型] = [变量] { }

let some_number = Some(5);
if let Some(3) = some_number {
    println!("{}", 3);
} else if let Some(x) = some_number {
    println!("{}", x);
} else {
    println!("None");
}
